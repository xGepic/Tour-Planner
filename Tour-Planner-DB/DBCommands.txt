CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

CREATE TABLE TourLog (
	TourLogID uuid DEFAULT uuid_generate_v4(),
	Tourname varchar(20),
	TourDateAndTime DATE,
	TourComment varchar(150),
	TourDifficulty int,
	TourTimeInMin int,
	TourRating int
)
